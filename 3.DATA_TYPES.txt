Strings
	immutable


Lists ----- []
	mutable
	.append()
	.insert(val, pos)
	.extend()
	.pop() ----- remove last element
	.pop(x) ----- remove xth position element
	.remove(x) ----- remove particular element


Tuples ----- ()
	immutable so Hashable
	tuple(list) ----- convert list to tuple


Sets ----- set()
	unordered
	iterable
	mutable

	dict({'1':2})
	.get()
	dict([(1, 'geeks'), (2, 'rishabh')])
